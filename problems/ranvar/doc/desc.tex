\feladatpp{valváltozó (ranvar)}{
Adott egy valségi változó, az értékkészletével és a kapcsolódó valségekkel:
\begin{gather*}
v_1,\ldots, v_n \\
p_1,\ldots,p_n
\end{gather*}
Számoljuk ki a várható értékét ($\mu$) és a szórását($\sigma$)!
}{
%input
\begin{gather*}
v_1,\ldots, v_n \\
p_1,\ldots,p_n
\end{gather*}
}{
%output
$\mu\ \ \sigma$
}{
%megjegyzés
$0\le n\le 100\ $
}{3}{5}
